  <script>
    const DATA_URL = 'https://script.google.com/macros/s/AKfycbzqf-oxbyV-SsQaUZj_C33xSQMX7h4V-g5Ws8EcXQ49nOYpn1_VWVwNhjnj3yd9PD_Syg/exec';

    function formatoPesos(valor) {
      return '$ ' + Number(valor).toLocaleString('es-AR', {
        minimumFractionDigits: 2,
        maximumFractionDigits: 2
      });
    }

    fetch(DATA_URL)
      .then(res => res.json())
      .then(d => {
        if (d.error) {
          document.getElementById('periodo').innerText = d.error;
          return;
        }
        document.getElementById('periodo').innerText   = d.periodo;
        document.getElementById('ingresos').innerText  = formatoPesos(d.ingresos);
        document.getElementById('egresos').innerText   = formatoPesos(d.egresos);
        document.getElementById('resultado').innerText = formatoPesos(d.resultado);
        document.getElementById('btnPdf').href =
          'https://drive.google.com/file/d/' + d.pdfId + '/view';
        document.getElementById('footer').innerHTML =
          'Fuente: José Luis Calvo<br>Última actualización: ' + d.fechaActualizacion;
      })
      .catch(() => {
        document.getElementById('periodo').innerText = 'Error cargando datos';
      });
  </script>
